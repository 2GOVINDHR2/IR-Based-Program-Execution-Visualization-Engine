def linear_search(arr, target):
    i = 0

    while i < len(arr):
        if arr[i] == target:
            return i
        i = i + 1

    return -1


# Input
n = int(input("Enter size (1 < n < 10): "))

if 1 < n < 10:
    arr = list(map(int, input("Enter elements: ").split()))

    if len(arr) != n:
        print("Invalid input length")
    else:
        target = int(input("Enter target: "))
        print("Result Index:", linear_search(arr, target))
else:
    print("Invalid size")
